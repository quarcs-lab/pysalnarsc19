```python
def read_csv(file_name):
    newlines = []
    with open(file_name, 'r') as f:
        reader = csv.reader(f)
        for line in reader:
            newlines.append(line)
    return newlines
    
```

```python
def get_attribute(data, attribute_name):
    header = data[0]
    idx = header.index(attribute_name)
    return [record[idx] for record in data[1:]]
```

```python
from math import radians, cos, sin, asin, sqrt

def haversine(lat1, lon1, lat2, lon2):

      R = 3959.87433 # this is in miles.  For Earth radius in kilometers use 6372.8 km

      dLat = radians(lat2 - lat1)
      dLon = radians(lon2 - lon1)
      lat1 = radians(lat1)
      lat2 = radians(lat2)

      a = sin(dLat/2)**2 + cos(lat1)*cos(lat2)*sin(dLon/2)**2
      c = 2*asin(sqrt(a))

      return R * c

# Usage
lon1 = -103.548851
lat1 = 32.0004311
lon2 = -103.6041946
lat2 = 33.374939

print(haversine(lat1, lon1, lat2, lon2))
```

```python
from math import inf
# find the nearest neighbor for an origin airport 
def nearest_airport(origin_airport, points):
    nnd = inf
    closest_j = inf
    i = origin_airport
    o_lat, o_lon = points[i]
    for j, point in enumerate(points):
        if i != j:
            d_lat, d_lon = point
            d = haversine(o_lat, o_lon, d_lat, d_lon)
            if d < nnd:
                nnd = d
                closest_j = j
    return closest_j, nnd
    
```
